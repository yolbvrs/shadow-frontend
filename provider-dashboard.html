document.addEventListener("DOMContentLoaded", async () => { const token = localStorage.getItem("token"); if (!token) { window.location.href = "login.html"; return; } const response = await fetch("https://shadow-backend-5xom.onrender.com/api/auth/user", { method: "GET", headers: { "Authorization": `Bearer ${token}` } }); const userData = await response.json(); if (userData.role !== "provider") { window.location.href = "customer-dashboard.html"; } document.getElementById("providerName").innerText = `Hello, ${userData.name}`; // âœ… Fetch job requests for the provider const jobResponse = await fetch(`https://shadow-backend-5xom.onrender.com/api/jobs/provider/${userData._id}`); const jobs = await jobResponse.json(); const jobList = document.getElementById("jobRequests"); jobList.innerHTML = ""; jobs.forEach(job => { const div = document.createElement("div"); div.innerHTML = `

Job Details: ${job.jobDetails}

Customer: ${job.customerId.name} (${job.customerId.email})

Status: ${job.status}

`; jobList.appendChild(div); }); });
